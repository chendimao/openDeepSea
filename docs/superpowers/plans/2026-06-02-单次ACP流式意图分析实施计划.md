# 单次 ACP 流式意图分析 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将群聊消息入口改为立即保存用户消息，并由单次只读 ACP 调用流式输出正文、末尾隐藏返回 intent JSON。

**Architecture:** 新增一个专用流式控制块解析器，负责从 ACP answer chunks 中剥离 `<openclaw_intent_json>` 内容并在完成后解析 intent。消息入口不再同步调用 ACP classifier；dispatcher 在普通全局群聊运行中强制使用 read-only prompt 契约，流式展示正文，完成后把 intent 写回源用户消息 metadata 并触发后续 route/task 决策。

**Tech Stack:** Node.js、TypeScript、Express、SQLite、node:test、ACP adapter streaming。

Create `packages/backend/src/acp-intent-stream.ts` with:

```ts
import type { MessageIntentResult, MessageIntentSuggestedAction } from './types.js';

const START_TAG = '<openclaw_intent_json>';
const END_TAG = '</openclaw_intent_json>';
const MAX_TAG_LENGTH = Math.max(START_TAG.length, END_TAG.length);

export interface AcpIntentStreamFilter {
  push(chunk: string): string;
  finish(): string;
  intentResult(): MessageIntentResult | null;
}

export function createAcpIntentStreamFilter(): AcpIntentStreamFilter {
  let visibleBuffer = '';
  let hiddenBuffer = '';
  let insideControlBlock = false;
  let parsedIntent: MessageIntentResult | null = null;

  const emitSafeVisible = (force = false): string => {
    if (!visibleBuffer) return '';
    if (force) {
      const out = visibleBuffer;
      visibleBuffer = '';
      return out;
    }
    const possibleStart = longestSuffixPrefix(visibleBuffer, START_TAG);
    const keep = Math.min(possibleStart, MAX_TAG_LENGTH - 1);
    if (visibleBuffer.length <= keep) return '';
    const out = visibleBuffer.slice(0, visibleBuffer.length - keep);
    visibleBuffer = visibleBuffer.slice(visibleBuffer.length - keep);
    return out;
  };

  const push = (chunk: string): string => {
    if (!chunk) return '';
    let output = '';
    let pending = chunk;
    while (pending) {
      if (!insideControlBlock) {
        visibleBuffer += pending;
        const startIndex = visibleBuffer.indexOf(START_TAG);
        if (startIndex < 0) {
          output += emitSafeVisible(false);
          pending = '';
          continue;
        }
        output += visibleBuffer.slice(0, startIndex);
        hiddenBuffer = '';
        visibleBuffer = visibleBuffer.slice(startIndex + START_TAG.length);
        insideControlBlock = true;
        pending = visibleBuffer;
        visibleBuffer = '';
        continue;
      }

      hiddenBuffer += pending;
      const endIndex = hiddenBuffer.indexOf(END_TAG);
      if (endIndex < 0) {
        pending = '';
        continue;
      }
      const rawIntent = hiddenBuffer.slice(0, endIndex);
      parsedIntent = parseAcpIntentControlBlock(rawIntent);
      pending = hiddenBuffer.slice(endIndex + END_TAG.length);
      hiddenBuffer = '';
      insideControlBlock = false;
    }
    return output;
  };

  return {
    push,
    finish() {
      if (insideControlBlock) {
        insideControlBlock = false;
        hiddenBuffer = '';
        return '';
      }
      return emitSafeVisible(true);
    },
    intentResult() {
      return parsedIntent;
    },
  };
}

export function parseAcpIntentControlBlock(raw: string): MessageIntentResult | null {
  let parsed: unknown;
  try {
    parsed = JSON.parse(raw.trim()) as unknown;
  } catch {
    return null;
  }
  if (!isRecord(parsed)) return null;
  const intent = readIntent(parsed.intent);
  const suggestedAction = readSuggestedAction(parsed.suggestedAction);
  if (!intent || !suggestedAction) return null;
  const confidence = typeof parsed.confidence === 'number' && Number.isFinite(parsed.confidence)
    ? Math.max(0, Math.min(1, parsed.confidence))
    : 1;
  const reason = typeof parsed.reason === 'string' && parsed.reason.trim()
    ? parsed.reason.trim()
    : 'ACP provided intent result';
  const signals = Array.isArray(parsed.signals)
    ? parsed.signals.filter((item): item is string => typeof item === 'string' && item.trim().length > 0).map((item) => item.trim())
    : [];
  return {
    intent,
    confidence,
    source: 'classifier',
    suggestedAction,
    reason,
    ...(signals.length > 0 ? { signals } : {}),
  };
}

function longestSuffixPrefix(value: string, token: string): number {
  const max = Math.min(value.length, token.length - 1);
  for (let length = max; length > 0; length -= 1) {
    if (value.slice(-length) === token.slice(0, length)) return length;
  }
  return 0;
}

function readIntent(value: unknown): MessageIntentResult['intent'] | null {
  if (
    value === 'chat' ||
    value === 'light_task' ||
    value === 'debugger' ||
    value === 'brainstorming' ||
    value === 'workflow'
  ) {
    return value;
  }
  return null;
}

function readSuggestedAction(value: unknown): MessageIntentSuggestedAction | null {
  if (
    value === 'reply_in_chat' ||
    value === 'create_light_task' ||
    value === 'start_debugger' ||
    value === 'start_brainstorming' ||
    value === 'start_workflow' ||
    value === 'ask_user'
  ) {
    return value;
  }
  return null;
}

function isRecord(value: unknown): value is Record<string, unknown> {
  return Boolean(value && typeof value === 'object' && !Array.isArray(value));
}
```
