# 任务四入口固定编队 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将任务创建后的主操作改为 `开始执行 / 头脑风暴 / 编写计划 / 子代理执行` 四入口，并让 `开始执行` 在启动时一次性固定完整编队和阶段计划。

**Architecture:** 新增后端 task action 入口，替代前端对旧 `task_execution.next_steps` 派发接口的直接依赖。`开始执行` 通过固定 roster 执行器锁定 executor/reviewer/acceptor 与阶段顺序；Superpowers 三个入口通过 ACP phase prompt 启动对应技能阶段，并把状态写入任务记录。前端抽出四入口组件，任务卡与任务详情共用同一套动作状态和 loading/blocked/failed 展示。

**Tech Stack:** TypeScript, Express, SQLite repos, React 18, TanStack Query, Vite, Node test runner.

```ts
import type { AgentRunStatus, RoomAgent, TaskActionKind, TaskActionStartResult, TaskWorkflowPlan } from './types.js';
import { agentRepo } from './repos/agents.js';
import { projectRepo } from './repos/projects.js';
import { roomAgentRepo, roomRepo } from './repos/rooms.js';
import { taskRepo } from './repos/tasks.js';
import { taskEventRepo } from './repos/task-events.js';
import { messageRepo } from './repos/messages.js';
import { wsHub } from './ws-hub.js';

export interface TaskActionAgentResult {
  status: AgentRunStatus | 'failed';
  content: string;
  error: string | null;
  runId?: string;
  messageId?: string;
}

export interface TaskActionRunAgentInput {
  agent: RoomAgent;
  prompt: string;
  taskId: string;
  sourceMessageId?: string | null;
}

export interface StartTaskActionInput {
  roomId: string;
  taskId: string;
  action: TaskActionKind;
  senderId?: string;
  senderName?: string;
  runAgent?: (input: TaskActionRunAgentInput) => Promise<TaskActionAgentResult>;
}

export async function startTaskAction(input: StartTaskActionInput): Promise<TaskActionStartResult> {
  const task = taskRepo.get(input.taskId);
  if (!task || task.room_id !== input.roomId) throw new Error('task not found');
  const room = roomRepo.get(input.roomId);
  if (!room) throw new Error('room not found');
  const project = projectRepo.get(room.project_id);
  if (!project) throw new Error('project not found');

  if (input.action === 'start_execution') {
    const workflow = ensureFixedRosterWorkflow(input.roomId, task.title);
    recordTaskActionEvent(input.roomId, input.taskId, input.action, 'running', { workflow });
    return {
      action: input.action,
      status: 'running',
      workflow,
      run_ids: [],
    };
  }

  return {
    action: input.action,
    status: 'blocked',
    run_ids: [],
    blocked_reason: 'task action phase is not implemented yet',
  };
}

function ensureFixedRosterWorkflow(roomId: string, taskTitle: string): TaskWorkflowPlan {
  const executor = selectOrAddAgentByRole(roomId, 'executor');
  const reviewer = selectOrAddAgentByRole(roomId, 'reviewer');
  const acceptor = selectOrAddAgentByRole(roomId, 'acceptor');
  return {
    mode: 'fixed_roster',
    entry_action: 'start_execution',
    locked: true,
    agents: [
      { agent_id: executor.agent_id, room_agent_id: executor.id, role: 'executor' },
      { agent_id: reviewer.agent_id, room_agent_id: reviewer.id, role: 'reviewer' },
      { agent_id: acceptor.agent_id, room_agent_id: acceptor.id, role: 'acceptor' },
    ],
    stages: [
      { id: 'execute', agent_ids: [executor.agent_id], parallel: false, goal: `实现任务：${taskTitle}` },
      { id: 'review', agent_ids: [reviewer.agent_id], parallel: false, goal: `审查任务实现：${taskTitle}` },
      { id: 'acceptance', agent_ids: [acceptor.agent_id], parallel: false, goal: `验收任务结果：${taskTitle}` },
    ],
  };
}

function selectOrAddAgentByRole(roomId: string, role: 'executor' | 'reviewer' | 'acceptor'): RoomAgent {
  const existing = roomAgentRepo.listByRoom(roomId).find((agent) => agent.workflow_role === role && agent.acp_enabled);
  if (existing) return existing;
  const fallbackId = role === 'executor' ? 'backend-executor' : role;
  const globalAgent = agentRepo.getByAgentId(fallbackId) ?? agentRepo.getByBuiltinKey(fallbackId);
  if (!globalAgent) throw new Error(`no ${role} agent available`);
  return roomAgentRepo.addFromGlobalAgent({ room_id: roomId, global_agent_id: globalAgent.id });
}

function recordTaskActionEvent(
  roomId: string,
  taskId: string,
  action: TaskActionKind,
  status: string,
  metadata: Record<string, unknown>,
): void {
  const task = taskRepo.get(taskId);
  const message = messageRepo.create({
    room_id: roomId,
    sender_type: 'system',
    sender_id: 'system',
    sender_name: 'System',
    content: `任务动作 ${action} 已进入 ${status}`,
    message_type: 'system',
    metadata: {
      task_id: taskId,
      event_type: 'task_updated',
      task_action: action,
      task_action_status: status,
      ...metadata,
    },
  });
  wsHub.broadcast(roomId, { type: 'message:new', roomId, message });
  if (task) {
    const event = taskEventRepo.create({
      room_id: roomId,
      task_id: taskId,
      type: 'task_updated',
      layer: 'timeline',
      payload: { action, status, message_id: message.id, ...metadata },
    });
    wsHub.broadcast(roomId, { type: 'task_event:new', roomId, event });
  }
}
```

```ts
async function defaultRunAgent(input: TaskActionRunAgentInput): Promise<TaskActionAgentResult> {
  let finalRunId: string | undefined;
  let finalMessageId: string | undefined;
  let finalStatus: AgentRunStatus | 'failed' = 'failed';
  let finalContent = '';
  let finalError: string | null = null;
  const room = roomRepo.get(input.agent.room_id);
  if (!room) throw new Error('room not found');
  const project = projectRepo.get(room.project_id);
  if (!project) throw new Error('project not found');

  await respondAsAgent({
    agent: input.agent,
    projectPath: project.path,
    roomId: room.id,
    prompt: input.prompt,
    taskId: input.taskId,
    sourceMessageId: input.sourceMessageId,
    onFinished: ({ run, message, status }) => {
      finalRunId = run.id;
      finalMessageId = message.id;
      finalStatus = status;
      finalContent = message.content;
      finalError = run.error;
    },
  });

  return {
    status: finalStatus,
    content: finalContent,
    error: finalError,
    runId: finalRunId,
    messageId: finalMessageId,
  };
}

async function runLockedRosterStages(input: {
  roomId: string;
  taskId: string;
  workflow: TaskWorkflowPlan;
  runAgent: (input: TaskActionRunAgentInput) => Promise<TaskActionAgentResult>;
}): Promise<{ status: 'completed' | 'failed' | 'blocked'; runIds: string[]; error: string | null }> {
  const runIds: string[] = [];
  const outputs: string[] = [];
  for (const stage of input.workflow.stages) {
    for (const agentId of stage.agent_ids) {
      const agent = roomAgentRepo.listByRoom(input.roomId).find((candidate) => candidate.agent_id === agentId);
      if (!agent) return { status: 'blocked', runIds, error: `locked roster agent missing: ${agentId}` };
      const result = await input.runAgent({
        agent,
        taskId: input.taskId,
        prompt: buildLockedStagePrompt(stage.id, stage.goal, outputs),
      });
      if (result.runId) runIds.push(result.runId);
      outputs.push(`${agent.agent_id}: ${result.content}`);
      if (result.status !== 'completed') return { status: 'failed', runIds, error: result.error ?? `${agent.agent_id} failed` };
    }
  }
  return { status: 'completed', runIds, error: null };
}

function buildLockedStagePrompt(stageId: string, goal: string, previousOutputs: string[]): string {
  return [
    `固定编队阶段：${stageId}`,
    `阶段目标：${goal}`,
    '本任务已锁定完整 roster。你不能要求新增智能体；如果发现缺人，说明阻塞原因，不要输出新增 agent 的 task_execution。',
    '',
    previousOutputs.length > 0 ? `前序阶段输出：\\n${previousOutputs.join('\\n\\n')}` : '前序阶段输出：无',
  ].join('\\n');
}
```

```ts
async function runSuperpowersPhaseAction(input: {
  roomId: string;
  taskId: string;
  action: TaskActionKind;
  runAgent: (input: TaskActionRunAgentInput) => Promise<TaskActionAgentResult>;
}): Promise<TaskActionStartResult> {
  const prerequisiteError = validatePhasePrerequisite(input.action, input.taskId);
  if (prerequisiteError) {
    recordTaskActionEvent(input.roomId, input.taskId, input.action, 'blocked', { blocked_reason: prerequisiteError });
    return {
      action: input.action,
      status: 'blocked',
      run_ids: [],
      blocked_reason: prerequisiteError,
    };
  }
  const phase = actionToPhase(input.action);
  if (!phase) throw new Error(`unsupported action: ${input.action}`);
  const task = taskRepo.get(input.taskId);
  const room = roomRepo.get(input.roomId);
  if (!task || !room) throw new Error('task or room not found');
  const project = projectRepo.get(room.project_id);
  if (!project) throw new Error('project not found');
  const agents = roomAgentRepo.listByRoom(input.roomId);
  const planner = agents.find((agent) => agent.agent_id === 'planner') ?? selectOrAddAgentByRole(input.roomId, 'reviewer');
  const prompt = buildSuperpowersPhasePrompt(phase, {
    projectName: project.name,
    projectPath: project.path,
    room,
    task,
    agents,
    workflowContext: `任务动作入口：${input.action}`,
  });
  recordTaskActionEvent(input.roomId, input.taskId, input.action, 'running', { superpowers_phase: phase });
  const result = await input.runAgent({ agent: planner, taskId: input.taskId, prompt, sourceMessageId: task.source_message_id });
  const status = result.status === 'completed' ? 'completed' : 'failed';
  recordTaskActionEvent(input.roomId, input.taskId, input.action, status, {
    superpowers_phase: phase,
    run_id: result.runId,
    error: result.error,
    evidence: extractSuperpowersEvidence(result.content),
  });
  return {
    action: input.action,
    status,
    run_ids: result.runId ? [result.runId] : [],
    message_id: result.messageId,
  };
}

function extractSuperpowersEvidence(content: string): Record<string, unknown> | null {
  const match = content.match(/```json\\s*([\\s\\S]*?)```/u);
  if (!match) return null;
  try {
    const parsed = JSON.parse(match[1] ?? '{}') as unknown;
    if (!parsed || typeof parsed !== 'object' || Array.isArray(parsed)) return null;
    const superpowers = (parsed as { superpowers?: unknown }).superpowers;
    return superpowers && typeof superpowers === 'object' && !Array.isArray(superpowers)
      ? superpowers as Record<string, unknown>
      : null;
  } catch {
    return null;
  }
}
```

```tsx
import { ClipboardList, Loader2, PenLine, Play, Workflow } from 'lucide-react';

interface TaskActionDefinition {

  label: string;
  description: string;
  icon: typeof Play;

  { id: 'brainstorming', label: '头脑风暴', description: '澄清需求并产出 spec', icon: PenLine },

];

export function TaskActionStrip({
  states,
  onStartAction,
  compact = false,
}: {
  states: Partial<Record<TaskActionKind, TaskActionState>>;
  onStartAction: (action: TaskActionKind) => void;
  compact?: boolean;

  return (

      {ACTIONS.map((action) => {

        const status = state?.status ?? 'idle';
        const running = status === 'queued' || status === 'running';
        const Icon = action.icon;
        return (
          <button
            key={action.id}
            type="button"
            className={cn('task-action-button', `is-${status}`)}
            disabled={running}
            onClick={() => onStartAction(action.id)}
            title={state?.detail ?? action.description}
          >
            {running ? <Loader2 className="h-3.5 w-3.5 animate-spin" /> : <Icon className="h-3.5 w-3.5" />}
            <span>
              <strong>{action.label}</strong>
              {!compact && <small>{state?.detail ?? action.description}</small>}
            </span>
          </button>
        );
      })}
    </div>
  );
}
```
