# 真正 ACP 客户端服务端接入 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将 Codex、Claude Code、OpenCode 的执行主路径从 CLI JSON 解析升级为真正 ACP client/server，并把协议事件稳定写入 `AgentTimelineEvent`。

**Architecture:** 保留现有 `SessionAdapter` 上层接口，新增 protocol-first 适配层；协议可用时走 ACP SDK + provider server，失败时回退 legacy CLI。protocol client 只负责把 ACP session update 转成 answer delta 或 raw protocol event，dispatcher 使用真实 message/run/agent 上下文归一化为 `AgentTimelineEvent`。

**Tech Stack:** Node.js 20、TypeScript、Express、WebSocket、SQLite、`@agentclientprotocol/sdk`、Node `child_process`、现有 `node:test` 测试栈。

```ts
import type { AcpBackend } from '../types.js';

export type AcpProtocolMode = 'auto' | 'protocol' | 'legacy';

export interface AcpServerConfig {
  backend: AcpBackend;
  mode: AcpProtocolMode;
  command: string;
  args: string[];
  env?: Record<string, string>;
  transport: 'stdio';
  enabled: boolean;
}

type EnvLike = Partial<Record<string, string | undefined>>;

const DEFAULT_COMMANDS: Record<AcpBackend, string> = {
  codex: 'npx @zed-industries/codex-acp',
  claudecode: 'npx @agentclientprotocol/claude-agent-acp',
  opencode: 'opencode acp',
};

const COMMAND_ENV: Record<AcpBackend, string> = {
  codex: 'OPENCLAW_ACP_CODEX_COMMAND',
  claudecode: 'OPENCLAW_ACP_CLAUDECODE_COMMAND',
  opencode: 'OPENCLAW_ACP_OPENCODE_COMMAND',
};

export function parseAcpMode(value: string | undefined): AcpProtocolMode {
  return value === 'protocol' || value === 'legacy' || value === 'auto' ? value : 'auto';
}

export function splitCommand(raw: string): { command: string; args: string[] } {
  const parts = raw.match(/(?:[^\s"]+|"[^"]*")+/g)?.map((part) => part.replace(/^"|"$/g, '')) ?? [];
  const [command, ...args] = parts;
  return { command: command ?? raw, args };
}

export function getAcpServerConfig(backend: AcpBackend, env: EnvLike = process.env): AcpServerConfig {
  const mode = parseAcpMode(env.OPENCLAW_ACP_MODE);
  const configured = env[COMMAND_ENV[backend]]?.trim();
  const split = splitCommand(configured || DEFAULT_COMMANDS[backend]);
  return {
    backend,
    mode,
    command: split.command,
    args: split.args,
    transport: 'stdio',
    enabled: mode !== 'legacy',
  };
}
```

```ts
import type { AgentTimelineEvent } from '../types.js';
import { createTimelineEvent, normalizeKnownProviderEvent } from './timeline.js';

export function normalizeProtocolEvent(args: {
  messageId: string;
  runId: string;
  agentId: string;
  seq: number;
  provider: string;
  raw: Record<string, unknown>;
}): AgentTimelineEvent {
  const payload = record(args.raw.update) ?? record(args.raw.payload) ?? args.raw;
  const type = text(args.raw.type) ?? text(payload.type) ?? args.provider;
  const kind = text(payload.kind);

  if (kind === 'thinking' || /thinking|reasoning/i.test(type)) {
    const value = text(payload.delta) ?? text(payload.text) ?? text(payload.content);
    return createTimelineEvent({
      messageId: args.messageId,
      runId: args.runId,
      agentId: args.agentId,
      seq: args.seq,
      type: 'thinking',
      status: /completed|done/i.test(type) ? 'completed' : 'delta',
      title: '思考过程',
      payload: { text: value ?? '', encrypted: payload.encrypted === true },
      raw: args.raw,
    });
  }

  if (/tool_call_started|tool.*started|tool_use/i.test(type)) {
    const name = text(payload.name) ?? text(args.raw.name) ?? 'tool';
    return createTimelineEvent({
      messageId: args.messageId,
      runId: args.runId,
      agentId: args.agentId,
      seq: args.seq,
      type: 'tool_call',
      status: 'started',
      title: `调用工具 ${name}`,
      payload: { id: text(payload.id) ?? text(args.raw.id), name, input: payload.input ?? args.raw.input },
      raw: args.raw,
    });
  }

  if (/tool_result|tool_call_completed|tool.*completed|function_call_output/i.test(type)) {
    const name = text(payload.name) ?? text(args.raw.name) ?? 'tool_result';
    return createTimelineEvent({
      messageId: args.messageId,
      runId: args.runId,
      agentId: args.agentId,
      seq: args.seq,
      type: 'tool_result',
      status: 'completed',
      title: `工具结果 ${name}`,
      payload: { id: text(payload.id) ?? text(args.raw.id), name, output: payload.output ?? payload.content ?? args.raw.output },
      raw: args.raw,
    });
  }

  if (/command_output|stdout|stderr/i.test(type)) {
    const stream = text(payload.stream) ?? text(args.raw.stream);
    return createTimelineEvent({
      messageId: args.messageId,
      runId: args.runId,
      agentId: args.agentId,
      seq: args.seq,
      type: 'command_output',
      status: 'delta',
      title: '命令输出',
      payload: {
        command: text(payload.command) ?? text(args.raw.command),
        stdout: stream === 'stderr' ? undefined : text(payload.text) ?? text(args.raw.text),
        stderr: stream === 'stderr' ? text(payload.text) ?? text(args.raw.text) : undefined,
      },
      raw: args.raw,
    });
  }

  if (/patch|diff|edit|file/i.test(type)) {
    return normalizeKnownProviderEvent({
      messageId: args.messageId,
      runId: args.runId,
      agentId: args.agentId,
      seq: args.seq,
      provider: args.provider,
      raw: args.raw,
    });
  }

  if (/plan|next_steps/i.test(type)) {
    return normalizeKnownProviderEvent({
      messageId: args.messageId,
      runId: args.runId,
      agentId: args.agentId,
      seq: args.seq,
      provider: args.provider,
      raw: args.raw,
    });
  }

  return normalizeKnownProviderEvent({
    messageId: args.messageId,
    runId: args.runId,
    agentId: args.agentId,
    seq: args.seq,
    provider: args.provider,
    raw: args.raw,
  });
}

function record(value: unknown): Record<string, unknown> | null {
  return value && typeof value === 'object' && !Array.isArray(value) ? value as Record<string, unknown> : null;
}

function text(value: unknown): string | null {
  return typeof value === 'string' && value.trim() ? value : null;
}
```

```ts
import * as acp from '@agentclientprotocol/sdk';
import { Readable, Writable } from 'node:stream';

const SESSION_ID = 'fake-session-1';

class FakeAgent implements acp.Agent {
  private connection: acp.AgentSideConnection | null = null;

  setConnection(connection: acp.AgentSideConnection): void {
    this.connection = connection;
  }

  async initialize(params: acp.InitializeRequest): Promise<acp.InitializeResponse> {
    return {
      protocolVersion: params.protocolVersion,
      agentCapabilities: {
        loadSession: true,
      },
    } as acp.InitializeResponse;
  }

  async newSession(): Promise<acp.NewSessionResponse> {
    return { sessionId: SESSION_ID } as acp.NewSessionResponse;
  }

  async loadSession(): Promise<acp.LoadSessionResponse> {
    return { sessionId: SESSION_ID } as acp.LoadSessionResponse;
  }

  async prompt(params: acp.PromptRequest): Promise<acp.PromptResponse> {
    await this.connection?.sessionUpdate({
      sessionId: params.sessionId,
      update: { sessionUpdate: 'agent_message_chunk', content: { type: 'text', text: 'fake answer' } },
    } as acp.SessionNotification);
```

- [ ] **Step 5: Implement protocol-client with ACP SDK**

创建 `packages/backend/src/acp/protocol-client.ts`。该实现直接使用 `@agentclientprotocol/sdk` 的 `ClientSideConnection` 和 `ndJsonStream`，不要再手写 JSON-RPC 解析：

```ts
import * as acp from '@agentclientprotocol/sdk';
import { spawn } from 'node:child_process';
import { Readable, Writable } from 'node:stream';
import type { AcpBackend } from '../types.js';
import type { AcpStreamChunk } from './types.js';
import type { AcpServerConfig } from './protocol-registry.js';

class OpenClawAcpClient implements acp.Client {
  constructor(private readonly onChunk: (chunk: AcpStreamChunk) => void) {}

  async sessionUpdate(params: acp.SessionNotification): Promise<void> {
    const update = params.update as Record<string, unknown>;
    if (update.sessionUpdate === 'agent_message_chunk') {
      const content = update.content as { type?: string; text?: string } | undefined;
      if (content?.type === 'text' && content.text) {
        this.onChunk({ stream: 'stdout', channel: 'answer', text: content.text });
        return;
      }
    }
    this.onChunk({
      stream: 'stdout',
      channel: 'event',
      text: '',
      rawType: typeof update.sessionUpdate === 'string' ? update.sessionUpdate : undefined,
      rawEvent: { type: 'session/update', ...params as unknown as Record<string, unknown> },
    });
  }

  async requestPermission(): Promise<acp.RequestPermissionResponse> {
    return { outcome: { outcome: 'selected', optionId: 'allow' } } as acp.RequestPermissionResponse;
  }

  async readTextFile(params: acp.ReadTextFileRequest): Promise<acp.ReadTextFileResponse> {
    const fs = await import('node:fs/promises');
    const url = new URL(params.uri);
    return { content: await fs.readFile(url, 'utf8') } as acp.ReadTextFileResponse;
  }

  async writeTextFile(params: acp.WriteTextFileRequest): Promise<acp.WriteTextFileResponse> {
    const fs = await import('node:fs/promises');
    const url = new URL(params.uri);
    await fs.writeFile(url, params.content);
    return {} as acp.WriteTextFileResponse;
  }
}

export async function invokeAcpProtocol(args: {
  backend: AcpBackend;
  projectPath: string;
  sessionId: string | null;
  prompt: string;
  server: AcpServerConfig;
  onChunk: (chunk: AcpStreamChunk) => void;
  onSession?: (sessionId: string) => void;
  signal?: AbortSignal;
}): Promise<{ exitCode: number; sessionId: string | null; stderr: string }> {
  return new Promise((resolve) => {
    const child = spawn(args.server.command, args.server.args, {
      cwd: args.projectPath,
      env: process.env,
      stdio: ['pipe', 'pipe', 'pipe'],
    });
    let stderr = '';
    let sessionId = args.sessionId;
    child.stderr?.setEncoding('utf8');
    child.stderr?.on('data', (data: string) => {
      stderr += data;
      args.onChunk({ stream: 'stderr', text: data });
    });

    child.on('error', (error) => resolve({ exitCode: -1, sessionId, stderr: error.message }));
    child.on('close', (code) => resolve({ exitCode: code ?? 0, sessionId, stderr }));
    args.signal?.addEventListener('abort', () => child.kill('SIGTERM'));

    void (async () => {
      if (!child.stdin || !child.stdout) {
        resolve({ exitCode: -1, sessionId, stderr: 'failed to open ACP stdio streams' });
        return;
      }
      const input = Writable.toWeb(child.stdin) as WritableStream<Uint8Array>;
      const output = Readable.toWeb(child.stdout) as ReadableStream<Uint8Array>;
      const stream = acp.ndJsonStream(input, output);
      const client = new OpenClawAcpClient(args.onChunk);
      const connection = new acp.ClientSideConnection(() => client, stream);
      const init = await connection.initialize({
        protocolVersion: acp.PROTOCOL_VERSION,
        clientCapabilities: {
          fs: { readTextFile: true, writeTextFile: true },
          terminal: true,
        },
      } as acp.InitializeRequest);
      const session = args.sessionId && init.agentCapabilities?.loadSession
        ? await connection.loadSession({ sessionId: args.sessionId, cwd: args.projectPath, mcpServers: [] } as acp.LoadSessionRequest)
        : await connection.newSession({ cwd: args.projectPath, mcpServers: [] } as acp.NewSessionRequest);
      sessionId = session.sessionId;
      args.onSession?.(sessionId);
      const prompt = [{ type: 'text', text: args.prompt }];
      await connection.prompt({ sessionId, prompt } as acp.PromptRequest);
      await connection.closeSession?.({ sessionId } as acp.CloseSessionRequest);
    })().catch((error: unknown) => {
      stderr += error instanceof Error ? error.message : String(error);
      child.kill('SIGTERM');
    });
  });
}
```

如果 SDK 类型名称和当前安装版本略有差异，以 `node_modules/@agentclientprotocol/sdk` 导出的类型为准做等价调整，但必须保留这四个行为：`ClientSideConnection`、`ndJsonStream`、`initialize`、`newSession/loadSession`、`prompt`。
