```ts
export async function listPlatformSkillAggregates(): Promise<PlatformSkillAggregate[]> {
  const skillsByProvider = await Promise.all(PLATFORM_PROVIDERS.map(async (provider) => ({
    provider,
    skills: await listPlatformSkills(provider),
  })));
  const byName = new Map<string, PlatformSkillAggregate>();

  for (const { provider, skills } of skillsByProvider) {
    for (const skill of skills) {
      const current = byName.get(skill.name) ?? createEmptyAggregate(skill.name);
      current.installations[provider] = skill;
      current.installModes[provider] = skill.installMode;
      current.description ??= skill.description;
      current.lastModifiedAt = maxTimestamp(current.lastModifiedAt, skill.lastModifiedAt);
      for (const issue of skill.issues) {
        current.issues.push({ provider, message: issue });
      }
      byName.set(skill.name, current);
    }
  }

  const aggregates = [...byName.values()];
  for (const aggregate of aggregates) {
    aggregate.providers = PLATFORM_PROVIDERS.filter((provider) => aggregate.installations[provider]);
    aggregate.missingProviders = PLATFORM_PROVIDERS.filter((provider) => !aggregate.installations[provider]);
    aggregate.valid = aggregate.providers.length > 0 && aggregate.issues.length === 0;
  }
  return aggregates.sort((a, b) => a.name.localeCompare(b.name));
}
```

Add these helpers near `readPlatformSkill()`:

```ts
function createEmptyAggregate(name: string): PlatformSkillAggregate {
  return {
    name,
    displayName: name,
    description: null,
    providers: [],
    missingProviders: [],
    installations: {},
    installModes: {},
    valid: false,
    issues: [],
    lastModifiedAt: null,
  };
}

function maxTimestamp(current: number | null, next: number | null): number | null {
  if (current === null) return next;
  if (next === null) return current;
  return Math.max(current, next);
}
```

    if (query) {

      if (!haystack.includes(query)) return false;
