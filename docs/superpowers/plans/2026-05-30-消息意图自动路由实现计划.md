# 消息意图自动路由 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现用户发送一条消息后，系统自动识别普通聊天、轻量任务、debugger、brainstorming、workflow，并在高置信时自动分流、低置信时展示确认入口。

**Architecture:** 新增后端 `MessageIntentRouter`，职责只判断“消息进入什么模式”，现有 `routeMessage()` 继续判断“消息归属哪个任务”。分类结果写入 `message.metadata.intent_result`，前端解析并在消息气泡中展示意图提示/低置信确认卡片；ACP / Superpowers 后续在实现阶段按意图承接，不让入口分类依赖 ACP 原始输出。

**Tech Stack:** Node.js `node:test`、TypeScript、Express、SQLite repos、React 18、Vite、现有 `MessageMetadata` / `MessageContent` / `RoomPage` 组件体系。

import type { MessageIntent, MessageIntentResult, MessageIntentSuggestedAction, RouteResult } from './types.js';

interface ClassifyMessageIntentInput {
  message: string;
}

export type MessageIntentClassifierInvoker = (prompt: string) => Promise<string>;

interface IntentRule {
  intent: MessageIntent;
  suggestedAction: MessageIntentSuggestedAction;
  reason: string;
  patterns: RegExp[];
  strongPatterns?: RegExp[];
}

const TASK_INTENTS = new Set<MessageIntent>(['light_task', 'debugger', 'brainstorming', 'workflow']);
const AUTO_CONFIDENCE_THRESHOLD = 0.85;

const rules: IntentRule[] = [
  {
    intent: 'debugger',
    suggestedAction: 'start_debugger',
    reason: '消息表达了排查、复现或修复问题的意图',
    patterns: [
      /为什么.+(没有|没|不|失败|报错|异常|变化)/u,
      /报错|错误|异常|失败|崩溃|无法|不能|没有变化/u,
      /debug|debugger|bug|复现|根因|排查|修复/u,
      /测试失败|构建失败|页面.*没有.*变化/u,
    ],
    strongPatterns: [/为什么|根因|复现|修复|没有变化/u],
  },
  {
    intent: 'brainstorming',
    suggestedAction: 'start_brainstorming',
    reason: '消息表达了头脑风暴、方案探索或交互设计讨论意图',
    patterns: [
      /头脑风暴|brainstorm/i,
      /重新优化.*(交互|布局|显示|样式|逻辑)/u,
      /给.*(方案|思路|选择)|讨论.*(方案|设计)|先.*设计/u,
      /先别实现|不要实现|只做方案|只做分析/u,
    ],
    strongPatterns: [/头脑风暴|brainstorm|方案|设计/u],
  },
  {
    intent: 'workflow',
    suggestedAction: 'start_workflow',
    reason: '消息表达了正式实现、完整闭环或多阶段交付意图',
    patterns: [
      /完整闭环|M1\s*到\s*M4|m1\s*到\s*m4/iu,
      /实现|开发|完成这个功能|接入|落地/u,
      /浏览器.*测试|实际测试|代码审查|code review|提交|commit/i,
      /workflow|工作流|writing-plans|verification/u,
    ],
    strongPatterns: [/完整闭环|M1\s*到\s*M4|浏览器.*测试|代码审查|提交|commit/iu],
  },
  {
    intent: 'light_task',
    suggestedAction: 'create_light_task',
    reason: '消息表达了小范围、低风险的轻量任务意图',
    patterns: [
      /轻量任务|临时插入|小改|改一下|调整一下|优化一下/u,
      /默认主题|文案|按钮|样式|颜色|标题/u,
      /把.+改成|加一个|删掉|隐藏/u,
    ],
    strongPatterns: [/临时插入|小改|改一下|默认主题|文案/u],
  },
  {
    intent: 'chat',
    suggestedAction: 'reply_in_chat',
    reason: '消息更像普通聊天、问答或解释请求',
    patterns: [
      /解释一下|是什么|怎么看|为什么要|能否|可以吗/u,
      /谢谢|好的|确定|继续|收到/u,
      /分析一下(?!.*实现)/u,
    ],
    strongPatterns: [/解释一下|是什么|怎么看|可以吗/u],
  },
];

export function classifyMessageIntent(input: ClassifyMessageIntentInput): MessageIntentResult {
  const message = input.message.trim();
  if (!message) return buildResult('chat', 'reply_in_chat', 0.4, '空消息按普通聊天处理', []);

  const candidates = rules
    .map((rule) => scoreRule(rule, message))
    .filter((candidate) => candidate.score > 0)
    .sort((left, right) => {
      if (right.score !== left.score) return right.score - left.score;
      return intentPriority(right.intent) - intentPriority(left.intent);
    });

  if (candidates.length === 0) {
    return buildResult('chat', 'ask_user', 0.55, '消息缺少明确聊天、任务、排查或工作流信号', []);
  }

  const winner = resolveConflicts(candidates);
  const confidence = Math.min(0.98, 0.62 + winner.score * 0.12 + winner.strongHits * 0.08);
  const suggestedAction = confidence >= 0.6 ? winner.suggestedAction : 'ask_user';
  return buildResult(
    winner.intent,
    confidence >= 0.85 ? winner.suggestedAction : suggestedAction === 'reply_in_chat' ? 'reply_in_chat' : 'ask_user',
    confidence,
    winner.reason,
    winner.signals,
  );
}

export async function classifyMessageIntentWithClassifier(input: {
  message: string;
  classifier?: MessageIntentClassifierInvoker;
}): Promise<MessageIntentResult> {
  const ruleResult = classifyMessageIntent({ message: input.message });
  if (!shouldAskUserForIntent(ruleResult) || !input.classifier) return ruleResult;

  try {
    const output = await input.classifier(buildClassifierPrompt(input.message, ruleResult));
    const parsed = parseClassifierIntentResult(output);
    return parsed && parsed.confidence >= ruleResult.confidence ? parsed : ruleResult;
  } catch {
    return ruleResult;
  }
}

export function parseClassifierIntentResult(output: string): MessageIntentResult | null {
  const candidate = extractJsonObject(output);
  if (!candidate) return null;
  let parsed: unknown;
  try {
    parsed = JSON.parse(candidate);
  } catch {
    return null;
  }
  if (!isRecord(parsed)) return null;
  if (
    !isMessageIntent(parsed.intent) ||
    !isMessageIntentSuggestedAction(parsed.suggestedAction) ||
    typeof parsed.confidence !== 'number' ||
    !Number.isFinite(parsed.confidence) ||
    parsed.confidence < 0 ||
    parsed.confidence > 1 ||
    typeof parsed.reason !== 'string' ||
    !parsed.reason.trim()
  ) {
    return null;
  }
  const signals = Array.isArray(parsed.signals)
    ? parsed.signals.filter((signal): signal is string => typeof signal === 'string' && signal.trim().length > 0).slice(0, 8)
    : [];
  return {
    intent: parsed.intent,
    confidence: parsed.confidence,
    reason: parsed.reason.trim(),
    source: 'classifier',
    suggestedAction: parsed.suggestedAction,
    signals,
  };
}

export function shouldAskUserForIntent(intent: MessageIntentResult): boolean {
  return intent.suggestedAction === 'ask_user' || intent.confidence < AUTO_CONFIDENCE_THRESHOLD;
}

export function applyIntentToRouteResult(routeResult: RouteResult, intent: MessageIntentResult): RouteResult {
  if (!TASK_INTENTS.has(intent.intent)) return routeResult;
  if (shouldAskUserForIntent(intent)) return routeResult;
  if (routeResult.action !== 'ask_user') return routeResult;
  return {
    taskId: null,
    action: 'create_task',
    confidence: Math.max(routeResult.confidence, intent.confidence),
    reason: `消息意图为 ${intent.intent}：${intent.reason}`,
  };
}

export function buildIntentActivityContent(intent: MessageIntentResult): string {
  const confidence = Math.round(intent.confidence * 100);
  if (shouldAskUserForIntent(intent)) {
    return `无法确定消息类型：建议 ${intent.intent}，置信度 ${confidence}%。原因：${intent.reason}`;
  }
  return `已识别消息类型：${intent.intent}，置信度 ${confidence}%。原因：${intent.reason}`;
}

function buildClassifierPrompt(message: string, ruleResult: MessageIntentResult): string {
  return [
    '你是 OpenDeepSea 的只读消息意图分类器。',
    '只能输出一个 JSON 对象，不要解释，不要调用工具，不要创建任务，不要启动 workflow。',
    'intent 只能是 chat、light_task、debugger、brainstorming、workflow。',
    'suggestedAction 只能是 reply_in_chat、create_light_task、start_debugger、start_brainstorming、start_workflow、ask_user。',
    `用户消息：${message}`,
    `规则初判：${JSON.stringify(ruleResult)}`,
  ].join('\n');
}

function scoreRule(rule: IntentRule, message: string): {
  intent: MessageIntent;
  suggestedAction: MessageIntentSuggestedAction;
  reason: string;
  score: number;
  strongHits: number;
  signals: string[];
} {
  const signals = rule.patterns
    .filter((pattern) => pattern.test(message))
    .map((pattern) => pattern.source);
  const strongHits = (rule.strongPatterns ?? []).filter((pattern) => pattern.test(message)).length;
  return {
    intent: rule.intent,
    suggestedAction: rule.suggestedAction,
    reason: rule.reason,
    score: signals.length,
    strongHits,
    signals,
  };
}

function resolveConflicts<T extends { intent: MessageIntent }>(candidates: T[]): T {
  const intents = new Set(candidates.map((candidate) => candidate.intent));
  if (intents.has('debugger')) return candidates.find((candidate) => candidate.intent === 'debugger')!;
  if (intents.has('brainstorming')) return candidates.find((candidate) => candidate.intent === 'brainstorming')!;
  if (intents.has('workflow')) return candidates.find((candidate) => candidate.intent === 'workflow')!;
  return candidates[0]!;
}

function intentPriority(intent: MessageIntent): number {
  switch (intent) {
    case 'debugger':
      return 5;
    case 'brainstorming':
      return 4;
    case 'workflow':
      return 3;
    case 'light_task':
      return 2;
    case 'chat':
      return 1;
  }
}

function buildResult(
  intent: MessageIntent,
  suggestedAction: MessageIntentSuggestedAction,
  confidence: number,
  reason: string,
  signals: string[],
): MessageIntentResult {
  return {
    intent,
    confidence: Math.max(0, Math.min(1, confidence)),
    reason,
    source: 'rule',
    suggestedAction,
    signals,
  };
}

function extractJsonObject(output: string): string | null {
  const fenced = /```(?:json)?\s*([\s\S]*?)```/iu.exec(output);
  const candidate = fenced?.[1]?.trim() ?? output.trim();
  return candidate.startsWith('{') && candidate.endsWith('}') ? candidate : null;
}

function isMessageIntent(value: unknown): value is MessageIntent {
  return value === 'chat' || value === 'light_task' || value === 'debugger' || value === 'brainstorming' || value === 'workflow';
}

function isMessageIntentSuggestedAction(value: unknown): value is MessageIntentSuggestedAction {
  return value === 'reply_in_chat' ||
    value === 'create_light_task' ||
    value === 'start_debugger' ||
    value === 'start_brainstorming' ||
    value === 'start_workflow' ||
    value === 'ask_user';
}

function isRecord(value: unknown): value is Record<string, unknown> {
  return Boolean(value && typeof value === 'object' && !Array.isArray(value));
}

```ts
function sanitizeMessageIntentMetadata(value: Record<string, unknown>) {
  const intentResult = sanitizeMessageIntentResult(value.intent_result);
  return intentResult ? { intent_result: intentResult } : {};
}

function sanitizeMessageIntentResult(value: unknown): MessageIntentResult | null {
  if (!isRecord(value)) return null;
  if (
    !isMessageIntent(value.intent) ||
    !isMessageIntentSource(value.source) ||
    !isMessageIntentSuggestedAction(value.suggestedAction) ||
    typeof value.confidence !== 'number' ||
    !Number.isFinite(value.confidence) ||
    value.confidence < 0 ||
    value.confidence > 1 ||
    typeof value.reason !== 'string' ||
    !value.reason.trim()
  ) {
    return null;
  }

  const signals = Array.isArray(value.signals)
    ? value.signals.filter((signal): signal is string => typeof signal === 'string' && signal.trim().length > 0)
    : [];

  return {
    intent: value.intent,
    confidence: value.confidence,
    reason: value.reason,
    source: value.source,
    suggestedAction: value.suggestedAction,
    signals: signals.slice(0, 8),
  };
}

function isMessageIntent(value: unknown): value is MessageIntent {
  return typeof value === 'string' && messageIntents.has(value as MessageIntent);
}

function isMessageIntentSource(value: unknown): value is MessageIntentSource {
  return typeof value === 'string' && messageIntentSources.has(value as MessageIntentSource);
}

function isMessageIntentSuggestedAction(value: unknown): value is MessageIntentSuggestedAction {
  return typeof value === 'string' && messageIntentSuggestedActions.has(value as MessageIntentSuggestedAction);
}
```
