# 聊天富文本编辑器与附件上传 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将聊天室消息发送区从 textarea 升级为支持整体 mention token、图片/文件预览和发送时 multipart 上传的富文本编辑器。

**Architecture:** 后端在现有 `POST /rooms/:roomId/messages` 上兼容 JSON 与 multipart：JSON 保持现状，multipart 保存附件后把元数据写入 `messages.metadata.attachments`。前端拆出 `RichMessageComposer`，正文状态使用 `ComposerNode[]`，附件状态独立管理；`RoomPage` 只负责派发发送、创建任务和消息刷新。

**Tech Stack:** Node.js、Express、multer、SQLite、React 18、TypeScript、Vite、Tailwind CSS、lucide-react、sonner、node:test。

```ts
import multer from 'multer';
import { mkdir, unlink } from 'node:fs/promises';
import { extname, join } from 'node:path';
import { fileURLToPath } from 'node:url';
import { nanoid } from 'nanoid';
import type { MessageAttachmentMetadata } from './types.js';

const __dirname = fileURLToPath(new URL('.', import.meta.url));

export const MAX_MESSAGE_FILES = 5;
export const MAX_MESSAGE_FILE_SIZE_BYTES = 10 * 1024 * 1024;
export const messageUploadDir = join(__dirname, '..', 'data', 'uploads', 'messages');
export const messageUploadRoute = '/uploads/messages';

const allowedExtensionPattern = /^[a-z0-9]{1,12}$/i;

export async function ensureMessageUploadDir(): Promise<void> {
  await mkdir(messageUploadDir, { recursive: true });
}

export function safeUploadFileName(originalName: string): string {
  const rawExt = extname(originalName).replace(/^\./, '').toLowerCase();
  const ext = allowedExtensionPattern.test(rawExt) ? rawExt : 'bin';
  return `${Date.now()}-${nanoid(12)}.${ext}`;
}

export const messageUpload = multer({
  storage: multer.diskStorage({
    destination: (_req, _file, cb) => {
      cb(null, messageUploadDir);
    },
    filename: (_req, file, cb) => {
      cb(null, safeUploadFileName(file.originalname));
    },
  }),
  limits: {
    files: MAX_MESSAGE_FILES,
    fileSize: MAX_MESSAGE_FILE_SIZE_BYTES,
  },
});

export function buildAttachmentMetadata(file: Express.Multer.File): MessageAttachmentMetadata {
  return {
    id: nanoid(16),
    name: file.originalname,
    mimeType: file.mimetype || 'application/octet-stream',
    size: file.size,
    url: `${messageUploadRoute}/${file.filename}`,
    isImage: Boolean(file.mimetype?.startsWith('image/')),
  };
}

export async function cleanupUploadedFiles(files: Express.Multer.File[] | undefined): Promise<void> {
  if (!files || files.length === 0) return;
  await Promise.allSettled(files.map((file) => unlink(file.path)));
}
```

```ts
  content: z.string().min(1),
  sender_id: z.string().default('user'),
  sender_name: z.string().optional(),
  mentions: z.array(z.string()).optional(),

router.post('/rooms/:roomId/messages', (req, res, next) => {
  if (req.is('multipart/form-data')) {
    messageUpload.array('files', 5)(req, res, (err) => {
      if (err) return next(err);
      void handleMultipartMessage(req, res).catch(next);
    });
    return;
  }
  void handleJsonMessage(req, res).catch(next);

async function handleJsonMessage(req: Request, res: Response): Promise<void> {
  const parsed = jsonMessageSchema.safeParse(req.body);
  if (!parsed.success) {
    res.status(400).json({ error: parsed.error.flatten() });
    return;
  }

  const userMsg = createAndDispatchUserMessage({
    roomId: req.params.roomId,
    senderId: parsed.data.sender_id,
    senderName: parsed.data.sender_name,
    content: parsed.data.content,
    mentions: parsed.data.mentions,
  });

  res.status(201).json(userMsg);
}

async function handleMultipartMessage(req: Request, res: Response): Promise<void> {
  const files = Array.isArray(req.files) ? req.files as Express.Multer.File[] : [];

  try {
    const parsed = multipartMessageSchema.safeParse(req.body);
    if (!parsed.success) {
      await cleanupUploadedFiles(files);
      res.status(400).json({ error: parsed.error.flatten() });
      return;
    }

    const content = parsed.data.content.trim();
    if (!content && files.length === 0) {
      res.status(400).json({ error: 'content or files is required' });
      return;
    }

    const mentions = parseMultipartMentions(parsed.data.mentions);
    if (!mentions.ok) {
      await cleanupUploadedFiles(files);
      res.status(400).json({ error: mentions.error });
      return;
    }

    const metadata: MessageMetadata = {
      attachments: files.map(buildAttachmentMetadata),
    };

    const userMsg = createAndDispatchUserMessage({
      roomId: req.params.roomId,
      senderId: parsed.data.sender_id,
      senderName: parsed.data.sender_name,
      content,
      mentions: mentions.value,
      metadata,
    });

    res.status(201).json(userMsg);
  } catch (err) {
    await cleanupUploadedFiles(files);
    throw err;
  }
}

function parseMultipartMentions(value: string | undefined): { ok: true; value?: string[] } | { ok: false; error: string } {
  if (!value) return { ok: true, value: undefined };
  try {
    const parsed = JSON.parse(value) as unknown;
    if (!Array.isArray(parsed) || !parsed.every((item) => typeof item === 'string')) {
      return { ok: false, error: 'mentions must be a JSON string array' };
    }
    return { ok: true, value: parsed };
  } catch {
    return { ok: false, error: 'mentions must be valid JSON' };
  }
}

function createAndDispatchUserMessage(input: {
  roomId: string;
  senderId: string;
  senderName?: string;
  content: string;
  mentions?: string[];
  metadata?: MessageMetadata;
}) {
  const userMsg = messageRepo.create({
    room_id: input.roomId,
    sender_type: 'user',
    sender_id: input.senderId,
    sender_name: input.senderName ?? 'You',
    content: input.content,
    message_type: 'text',
    metadata: input.metadata,
  });
  wsHub.broadcast(input.roomId, { type: 'message:new', roomId: input.roomId, message: userMsg });
  const agents = roomAgentRepo.listByRoom(input.roomId);
  const mentionedAgentRoomIds = resolveMentionedAgentRoomIds({
    content: input.content,
    agents,
    explicitRoomAgentIds: input.mentions,
  });
  void dispatchUserMessage({
    roomId: input.roomId,
    userMessage: userMsg,
    mentionedAgentRoomIds,
  });
  return userMsg;
}
```

```ts
import type { RoomAgent } from './types';

export const MAX_MESSAGE_FILES = 5;
export const MAX_MESSAGE_FILE_SIZE_BYTES = 10 * 1024 * 1024;

export type ComposerNode =
  | { type: 'text'; text: string }
  | {
      type: 'mention';
      roomAgentId: string;
      agentId: string;
      agentName: string;
      label: string;
    };

export interface PendingAttachment {
  id: string;
  file: File;
  name: string;
  mimeType: string;
  size: number;
  isImage: boolean;
  previewUrl: string | null;
}

export interface SerializedComposer {
  content: string;
  mentions: string[];
}

const typedMentionPattern = /@([\p{L}\p{N}_.-]+)/gu;

export function createEmptyComposerNodes(): ComposerNode[] {
  return [{ type: 'text', text: '' }];
}

export function serializeComposerNodes(nodes: ComposerNode[]): SerializedComposer {
  const mentions: string[] = [];
  const content = nodes.map((node) => {
    if (node.type === 'text') return node.text;
    mentions.push(node.roomAgentId);
    return `@${node.agentName}`;
  }).join('');

  return {
    content,
    mentions: Array.from(new Set(mentions)),
  };
}

export function findUniqueAgentMention(token: string, agents: RoomAgent[]): RoomAgent | null {
  const normalized = token.toLowerCase();
  const matches = agents.filter((agent) =>
    agent.agent_name.toLowerCase() === normalized || agent.agent_id.toLowerCase() === normalized,
  );
  return matches.length === 1 ? matches[0] : null;
}

export function normalizeTypedMentions(nodes: ComposerNode[], agents: RoomAgent[]): ComposerNode[] {
  const normalized: ComposerNode[] = [];

  for (const node of nodes) {
    if (node.type === 'mention') {
      normalized.push(node);
      continue;
    }

    let lastIndex = 0;
    let match: RegExpExecArray | null;
    typedMentionPattern.lastIndex = 0;
    while ((match = typedMentionPattern.exec(node.text)) !== null) {
      const agent = findUniqueAgentMention(match[1], agents);
      if (!agent) continue;
      if (match.index > lastIndex) {
        normalized.push({ type: 'text', text: node.text.slice(lastIndex, match.index) });
      }
      normalized.push({
        type: 'mention',
        roomAgentId: agent.id,
        agentId: agent.agent_id,
        agentName: agent.agent_name,
        label: `@${agent.agent_name}`,
      });
      lastIndex = match.index + match[0].length;
    }

    if (lastIndex < node.text.length) {
      normalized.push({ type: 'text', text: node.text.slice(lastIndex) });
    }
  }

  return mergeAdjacentTextNodes(normalized.length > 0 ? normalized : createEmptyComposerNodes());
}

export function mergeAdjacentTextNodes(nodes: ComposerNode[]): ComposerNode[] {
  const merged: ComposerNode[] = [];
  for (const node of nodes) {
    const previous = merged[merged.length - 1];
    if (node.type === 'text' && previous?.type === 'text') {
      previous.text += node.text;
      continue;
    }
    merged.push({ ...node });
  }
  return merged.length > 0 ? merged : createEmptyComposerNodes();
}

export function formatFileSize(size: number): string {
  if (size < 1024) return `${size} B`;
  if (size < 1024 * 1024) return `${(size / 1024).toFixed(1)} KB`;
  return `${(size / 1024 / 1024).toFixed(1)} MB`;
}

export function validatePendingFiles(existingCount: number, files: File[]): string | null {
  if (existingCount + files.length > MAX_MESSAGE_FILES) return '单条消息最多允许 5 个文件';
  const oversized = files.find((file) => file.size > MAX_MESSAGE_FILE_SIZE_BYTES);
  if (oversized) return `${oversized.name} 超过 10MB 限制`;
  return null;
}
```

```ts
async function request<T>(path: string, init: RequestInit = {}): Promise<T> {
  const isFormData = init.body instanceof FormData;
  const headers = {
    ...(isFormData ? {} : { 'Content-Type': 'application/json' }),
    ...(init.headers ?? {}),
  };

  const res = await fetch(`${BASE}${path}`, {
    ...init,
    headers,
  });
  if (!res.ok) {
    const text = await res.text().catch(() => res.statusText);
    throw new Error(`${res.status}: ${text}`);
  }
  if (res.status === 204) return undefined as T;
  return (await res.json()) as T;
}
```

```tsx
  const canSend = !disabled && !sending && (serialized.content.trim().length > 0 || attachments.length > 0);

  useEffect(() => {
    attachmentsRef.current = attachments;
  }, [attachments]);

  useEffect(() => {
    return () => {
      attachmentsRef.current.forEach((attachment) => {
        if (attachment.previewUrl) URL.revokeObjectURL(attachment.previewUrl);
      });
    };
  }, []);

  const syncFromDom = () => {
    const text = editorRef.current?.innerText ?? '';
    const nextNodes = normalizeTypedMentions([{ type: 'text', text }], agents);
    setNodes(nextNodes);
    const beforeCursor = getTextBeforeCursor(editorRef.current);
    const match = beforeCursor.match(/@([\p{L}\p{N}_.-]*)$/u);
    setMention(match ? { query: match[1] } : null);
  };

  const renderNodesToEditor = (nextNodes: ComposerNode[]) => {
    const editor = editorRef.current;
    if (!editor) return;
    editor.innerHTML = '';
    nextNodes.forEach((node) => {
      if (node.type === 'text') {
        editor.append(document.createTextNode(node.text));
        return;
      }
      const chip = document.createElement('span');
      chip.className = 'composer-mention-token';
      chip.contentEditable = 'false';
      chip.dataset.roomAgentId = node.roomAgentId;
      chip.textContent = node.label;
      editor.append(chip);
    });
    placeCursorAtEnd(editor);
  };

  const selectAgent = (agent: RoomAgent) => {
    const text = editorRef.current?.innerText ?? serialized.content;
    const replaced = text.replace(/@([\p{L}\p{N}_.-]*)$/u, `@${agent.agent_name} `);
    const nextNodes = normalizeTypedMentions([{ type: 'text', text: replaced }], agents);
    setNodes(nextNodes);
    setMention(null);
    window.requestAnimationFrame(() => renderNodesToEditor(nextNodes));
  };

  const addFiles = (files: File[]) => {
    if (files.length === 0) return;
    const error = validatePendingFiles(attachments.length, files);
    if (error) {
      toast.error(error);
      return;
    }
    setAttachments((current) => [
      ...current,
      ...files.map((file) => ({
        id: crypto.randomUUID(),
        file,
        name: file.name,
        mimeType: file.type || 'application/octet-stream',
        size: file.size,
        isImage: file.type.startsWith('image/'),
        previewUrl: file.type.startsWith('image/') ? URL.createObjectURL(file) : null,
      })),
    ]);
  };

  const removeAttachment = (id: string) => {
    setAttachments((current) => {
      const target = current.find((attachment) => attachment.id === id);
      if (target?.previewUrl) URL.revokeObjectURL(target.previewUrl);
      return current.filter((attachment) => attachment.id !== id);
    });
  };

  const submit = () => {
    const normalizedNodes = normalizeTypedMentions(nodes, agents);
    const nextSerialized = serializeComposerNodes(normalizedNodes);
    const content = nextSerialized.content.trim();
    if (content.startsWith('/task') && attachments.length > 0) {
      toast.error('/task 命令不能携带附件，请先移除附件');
      return;
    }
    if (!content && attachments.length === 0) return;

    onSend({
      content,
      mentions: nextSerialized.mentions.length > 0 ? nextSerialized.mentions : undefined,
      files: attachments.map((attachment) => attachment.file),
    });
  };

  const clear = () => {
    attachments.forEach((attachment) => {
      if (attachment.previewUrl) URL.revokeObjectURL(attachment.previewUrl);
    });
    setAttachments([]);
    setNodes(createEmptyComposerNodes());
    if (editorRef.current) editorRef.current.innerHTML = '';
  };

  useEffect(() => {
    if (!sending) return;
    setMention(null);
  }, [sending]);

  useEffect(() => {
    if (!didMountRef.current) {
      didMountRef.current = true;
      return;
    }
    clear();
  }, [resetKey]);

  return (
    <div className="composer-shell flex-shrink-0 px-4 py-3">
      <form
        onSubmit={(event) => {
          event.preventDefault();
          submit();
        }}
        className="relative space-y-2"
      >
        {mention && <AgentMentionMenu agents={agents} query={mention.query} onSelect={selectAgent} />}
        <div
          className={cn('composer-box rich-composer-box', disabled && 'opacity-60')}
          onDrop={(event) => {
            event.preventDefault();
            addFiles(Array.from(event.dataTransfer.files));
          }}
          onDragOver={(event) => event.preventDefault()}
        >
          {attachments.length > 0 && (
            <div className="composer-attachments">
              {attachments.map((attachment) => (
                <div key={attachment.id} className="composer-attachment">
                  {attachment.previewUrl ? (
                    <img src={attachment.previewUrl} alt={attachment.name} />
                  ) : (
                    <Paperclip className="h-4 w-4" />
                  )}
                  <div className="min-w-0 flex-1">
                    <div className="truncate text-[12px] font-medium">{attachment.name}</div>
                    <div className="text-[10.5px] text-[var(--color-muted)]">
                      {formatFileSize(attachment.size)} · {attachment.mimeType}
                    </div>
                  </div>
                  <button type="button" className="icon-glass-button h-7 w-7" onClick={() => removeAttachment(attachment.id)} aria-label={`移除 ${attachment.name}`}>
                    <X className="h-3.5 w-3.5" />
                  </button>
                </div>
              ))}
            </div>
          )}
          <div className="flex items-end gap-2">
            <input
              ref={fileInputRef}
              type="file"
              multiple
              className="hidden"
              onChange={(event) => {
                addFiles(Array.from(event.currentTarget.files ?? []));
                event.currentTarget.value = '';
              }}
            />
            <button type="button" className="icon-glass-button h-[40px] w-[40px]" onClick={() => fileInputRef.current?.click()} disabled={disabled} aria-label="上传图片或文件">
              <Paperclip className="h-4 w-4" />
            </button>
            <div
              ref={editorRef}
              contentEditable={!disabled}
              role="textbox"
              aria-multiline="true"
              data-placeholder={placeholder}
              className="rich-composer-editor"
              onInput={syncFromDom}
              onBlur={() => {
                const nextNodes = normalizeTypedMentions(nodes, agents);
                setNodes(nextNodes);
                renderNodesToEditor(nextNodes);
              }}
              onPaste={(event) => {
                const files = Array.from(event.clipboardData.files);
                if (files.length > 0) {
                  event.preventDefault();
                  addFiles(files);
                }
              }}
              onKeyDown={(event) => {
                if (event.key === 'Escape' && mention) {
                  event.preventDefault();
                  setMention(null);
                  return;
                }
                if (event.key === 'Enter' && !event.shiftKey) {
                  event.preventDefault();
                  submit();
                }
              }}
            />
            <Button type="submit" disabled={!canSend} className="h-[40px] w-[78px] flex-shrink-0 px-0">
              <Send className="h-3.5 w-3.5" /> 发送
            </Button>
          </div>
        </div>
        <p className="px-1 text-[11.5px] leading-relaxed text-[var(--color-fg-muted)]">{routingHint}</p>
      </form>
    </div>
  );
}

function getTextBeforeCursor(editor: HTMLDivElement | null): string {
  if (!editor) return '';
  const selection = window.getSelection();
  if (!selection || selection.rangeCount === 0) return editor.innerText;
  const range = selection.getRangeAt(0).cloneRange();
  range.selectNodeContents(editor);
  range.setEnd(selection.anchorNode ?? editor, selection.anchorOffset);
  return range.toString();
}

function placeCursorAtEnd(element: HTMLElement): void {
  element.focus();
  const range = document.createRange();
  range.selectNodeContents(element);
  range.collapse(false);
  const selection = window.getSelection();
  selection?.removeAllRanges();
  selection?.addRange(range);
}
```

```ts
import type { MessageAttachmentMetadata, MessageMetadata } from './types';

const emptyMessageMetadata: MessageMetadata = {
  attachments: [],
};

export function parseMessageMetadata(metadata: string | null): MessageMetadata {
  if (!metadata) return emptyMessageMetadata;
  try {
    const parsed = JSON.parse(metadata) as unknown;
    if (!isRecord(parsed)) return emptyMessageMetadata;
    const attachments = Array.isArray(parsed.attachments)
      ? parsed.attachments.filter(isMessageAttachmentMetadata)
      : [];
    return { attachments };
  } catch {
    return emptyMessageMetadata;
  }
}

function isMessageAttachmentMetadata(value: unknown): value is MessageAttachmentMetadata {
  if (!isRecord(value)) return false;
  return typeof value.id === 'string'
    && typeof value.name === 'string'
    && typeof value.mimeType === 'string'
    && typeof value.size === 'number'
    && typeof value.url === 'string'
    && typeof value.isImage === 'boolean';
}

function isRecord(value: unknown): value is Record<string, unknown> {
  return typeof value === 'object' && value !== null;
}
```
