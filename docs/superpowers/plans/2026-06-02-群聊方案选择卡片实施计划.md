# 群聊方案选择卡片 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将群聊头脑风暴消息中的推荐、备选和不推荐方案展示为可选择卡片，点击后只发送选择消息，不直接创建任务或启动 workflow。

**Architecture:** 前端先用结构化 metadata 渲染方案卡片；没有 metadata 时，用保守 Markdown 解析器识别明确的头脑风暴方案段落。卡片点击通过现有 `sendMessage` 链路发送普通用户消息，并携带 `brainstorming_option_selection` metadata；后端只保存 metadata 并继续走现有消息路由，由 planner/intent 决定下一步。

**Tech Stack:** React 18、TypeScript、Vite、Node `node:test`、Express、Zod、SQLite 消息仓储。

```ts
function sanitizeBrainstormingMetadata(value: Record<string, unknown>) {
  const options = sanitizeBrainstormingOptions(value.brainstorming_options);
  const selection = sanitizeBrainstormingOptionSelection(value.brainstorming_option_selection);
  return {
    ...(options.length > 0 ? { brainstorming_options: options } : {}),
    ...(selection ? { brainstorming_option_selection: selection } : {}),
  };
}

function sanitizeBrainstormingOptions(value: unknown): BrainstormingOption[] {
  if (!Array.isArray(value)) return [];
  return value
    .map(sanitizeBrainstormingOption)
    .filter((option): option is BrainstormingOption => option !== null)
    .slice(0, 6);
}

function sanitizeBrainstormingOption(value: unknown): BrainstormingOption | null {
  if (!isRecord(value)) return null;
  if (
    typeof value.id !== 'string' ||
    !value.id.trim() ||
    typeof value.title !== 'string' ||
    !value.title.trim() ||
    typeof value.summary !== 'string' ||
    !value.summary.trim() ||
    !isBrainstormingOptionMaturity(value.maturity)
  ) {
    return null;
  }

  return {
    id: value.id.trim().slice(0, 120),
    title: value.title.trim().slice(0, 120),
    summary: value.summary.trim().slice(0, 360),
    benefits: sanitizeStringList(value.benefits, 3, 180),
    risks: sanitizeStringList(value.risks, 3, 180),
    maturity: value.maturity,
    ...(typeof value.recommended === 'boolean' ? { recommended: value.recommended } : {}),
  };
}

function sanitizeBrainstormingOptionSelection(value: unknown): BrainstormingOptionSelection | null {
  if (!isRecord(value)) return null;
  if (
    typeof value.selected_option_id !== 'string' ||
    !value.selected_option_id.trim() ||
    typeof value.selected_option_title !== 'string' ||
    !value.selected_option_title.trim() ||
    !isBrainstormingOptionMaturity(value.selected_option_maturity) ||
    typeof value.source_message_id !== 'string' ||
    !value.source_message_id.trim() ||
    value.source_type !== 'brainstorming_option'
  ) {
    return null;
  }

  return {
    selected_option_id: value.selected_option_id.trim().slice(0, 120),
    selected_option_title: value.selected_option_title.trim().slice(0, 160),
    selected_option_maturity: value.selected_option_maturity,
    source_message_id: value.source_message_id.trim().slice(0, 120),
    source_type: 'brainstorming_option',
  };
}

function sanitizeStringList(value: unknown, limit: number, maxLength: number): string[] {
  if (!Array.isArray(value)) return [];
  return value
    .filter((item): item is string => typeof item === 'string')
    .map((item) => item.trim())
    .filter((item) => item.length > 0)
    .map((item) => item.slice(0, maxLength))
    .slice(0, limit);
}

function isBrainstormingOptionMaturity(value: unknown): value is BrainstormingOptionMaturity {
  return typeof value === 'string' && brainstormingOptionMaturities.has(value as BrainstormingOptionMaturity);
}
```

```ts
import type { BrainstormingOption, BrainstormingOptionMaturity, Message, MessageMetadata } from '../../lib/types';

const OPTION_HEADING_PATTERN = /^(推荐方案|备选轻量方案|备选方案|不推荐方案|方案\\s*[A-CＡ-Ｃ一二三1-3])[:：]\\s*(.+)$/;

export function getBrainstormingOptionsForMessage(
  message: Message,
  metadata: MessageMetadata,
): BrainstormingOption[] {
  if (message.sender_type !== 'agent') return [];
  if (metadata.brainstorming_options && metadata.brainstorming_options.length > 0) {
    return metadata.brainstorming_options;
  }
  return parseBrainstormingOptionsFromMarkdown(message.content);
}

export function parseBrainstormingOptionsFromMarkdown(content: string): BrainstormingOption[] {
  if (!content.includes('方案')) return [];

  const lines = content.split(/\\r?\\n/);
  const options: BrainstormingOption[] = [];
  let current: DraftOption | null = null;

  for (const rawLine of lines) {
    const line = rawLine.trim();
    const match = OPTION_HEADING_PATTERN.exec(line);
    if (match) {
      if (current) options.push(finalizeDraftOption(current));
      const title = normalizeOptionTitle(match[1]);
      current = {
        title,
        summaryParts: [stripMarkdown(match[2])],
        bullets: [],
      };
      continue;
    }

    if (!current) continue;
    if (!line) continue;

    const bullet = /^[-*]\\s+(.+)$/.exec(line);
    if (bullet) {
      current.bullets.push(stripMarkdown(bullet[1]));
      continue;
    }

    if (OPTION_HEADING_PATTERN.test(line)) continue;
    current.summaryParts.push(stripMarkdown(line));
  }

  if (current) options.push(finalizeDraftOption(current));
  return options.filter((option) => option.summary).slice(0, 6);
}

interface DraftOption {
  title: string;
  summaryParts: string[];
  bullets: string[];
}

function finalizeDraftOption(draft: DraftOption): BrainstormingOption {
  const maturity = inferMaturity(draft.title, draft.summaryParts.join(' '));
  const summary = compactText(draft.summaryParts.join(' '), 240);
  const benefits = draft.bullets.slice(0, 3).map((item) => compactText(item, 160));
  const risks = inferRisks(draft.title, draft.summaryParts.join(' '));
  return {
    id: createOptionId(draft.title, summary),
    title: draft.title,
    summary,
    benefits,
    risks,
    maturity,
    ...(draft.title.includes('推荐') && !draft.title.includes('不推荐') ? { recommended: true } : {}),
  };
}

function normalizeOptionTitle(value: string): string {
  const trimmed = value.replace(/\\s+/g, '').trim();
  if (trimmed.includes('推荐方案') && !trimmed.includes('不推荐')) return '推荐方案';
  if (trimmed.includes('备选轻量方案')) return '备选轻量方案';
  if (trimmed.includes('备选方案')) return '备选方案';
  if (trimmed.includes('不推荐方案')) return '不推荐方案';
  return value.trim();
}

function inferMaturity(title: string, text: string): BrainstormingOptionMaturity {
  if (title.includes('不推荐')) return 'exploratory';
  if (title.includes('轻量')) return 'actionable';
  if (/还需要|需要确认|边界|风险|下游|多轮/.test(text)) return 'boundary_needed';
  return title.includes('推荐') ? 'boundary_needed' : 'exploratory';
}

function inferRisks(title: string, text: string): string[] {
  if (title.includes('不推荐')) return ['该方案主要用于排除路径，不建议直接实现。'];
  if (/目录发送后仍不可用|第二个问题/.test(text)) return ['只能解决第一感知，目录下游语义仍需后续补齐。'];
  if (/需要确认|边界|目录注入/.test(text)) return ['仍需确认触发规则、目录注入深度或错误提示边界。'];
  return [];
}

function createOptionId(title: string, summary: string): string {
  const source = `${title}-${summary}`;
  const ascii = source
    .normalize('NFKD')
    .replace(/[^a-zA-Z0-9\\u4e00-\\u9fa5]+/g, '-')
    .replace(/^-+|-+$/g, '')
    .slice(0, 80);
  return ascii || 'brainstorming-option';
}

function stripMarkdown(value: string): string {
  return value
    .replace(/`([^`]+)`/g, '$1')
    .replace(/\\[([^\\]]+)\\]\\([^)]+\\)/g, '$1')
    .replace(/\\*\\*([^*]+)\\*\\*/g, '$1')
    .trim();
}

function compactText(value: string, maxLength: number): string {
  const normalized = value.replace(/\\s+/g, ' ').trim();
  return normalized.length > maxLength ? `${normalized.slice(0, maxLength - 1)}…` : normalized;
}
```

```tsx
import { Check, Lightbulb, MousePointerClick } from 'lucide-react';
import type { BrainstormingOption, BrainstormingOptionMaturity } from '../../lib/types';
import { cn } from '../../lib/utils';

interface BrainstormingOptionsPanelProps {
  sourceMessageId: string;
  options: BrainstormingOption[];
  selectedOptionIds: Set<string>;
  selecting: boolean;
  onSelect: (sourceMessageId: string, option: BrainstormingOption) => void;
}

export function BrainstormingOptionsPanel({
  sourceMessageId,
  options,
  selectedOptionIds,
  selecting,
  onSelect,
}: BrainstormingOptionsPanelProps): JSX.Element | null {
  if (options.length === 0) return null;

  return (
    <section className="brainstorming-options-panel" aria-label="头脑风暴方案选择">
      <div className="brainstorming-options-header">
        <Lightbulb className="h-3.5 w-3.5" strokeWidth={1.8} aria-hidden="true" />
        <span>可选方案</span>
      </div>
      <div className="brainstorming-options-grid">
        {options.map((option) => {
          const selected = selectedOptionIds.has(option.id);
          return (
            <article
              key={option.id}
              className={cn(
                'brainstorming-option-card',
                `is-${option.maturity}`,
                option.recommended && 'is-recommended',
                selected && 'is-selected',
              )}
            >
              <div className="brainstorming-option-topline">
                <strong>{option.title}</strong>
                <span>{maturityLabel(option.maturity)}</span>
              </div>
              <p>{option.summary}</p>
              {option.benefits.length > 0 && (
                <ul>
                  {option.benefits.map((benefit) => (
                    <li key={benefit}>{benefit}</li>
                  ))}
                </ul>
              )}
              {option.risks.length > 0 && (
                <div className="brainstorming-option-risk">{option.risks[0]}</div>
              )}
              <button
                type="button"
                className="brainstorming-option-action"
                disabled={selecting || selected}
                onClick={() => onSelect(sourceMessageId, option)}
              >
                {selected ? (
                  <>
                    <Check className="h-3.5 w-3.5" strokeWidth={1.9} aria-hidden="true" />
                    已选择
                  </>
                ) : (
                  <>
                    <MousePointerClick className="h-3.5 w-3.5" strokeWidth={1.9} aria-hidden="true" />
                    选择此方案
                  </>
                )}
              </button>
            </article>
          );
        })}
      </div>
    </section>
  );
}

function maturityLabel(maturity: BrainstormingOptionMaturity): string {
  switch (maturity) {
    case 'exploratory':
      return '需继续讨论';
    case 'boundary_needed':
      return '需确认边界';
    case 'actionable':
      return '可直接实现';
  }
}
```
