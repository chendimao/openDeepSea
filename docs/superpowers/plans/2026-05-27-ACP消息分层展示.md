# ACP 消息分层展示 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a layered ACP message UI where assistant text streams only into the chat body, process events render in a clean timeline, and protocol noise stays in a hidden debug monitor.

**Architecture:** Keep backend event persistence unchanged. Add a frontend timeline view-model layer that classifies `AgentTimelineEvent[]` into visible process events and hidden debug events, merges tool lifecycle events by `toolCallId`, and exposes a stable model to `AgentTimeline`. Update `AgentTimeline` to render the model instead of raw protocol events.

**Tech Stack:** React 18, TypeScript, Vite, Node test runner with `tsx`, existing `AgentTimeline` CSS in `packages/frontend/src/index.css`.

```ts
import type { AgentTimelineEvent } from '../../lib/types';

export interface AgentTimelineModel {
  visibleEvents: AgentTimelineEvent[];
  debugEvents: AgentTimelineEvent[];
  visibleCount: number;
}

const debugRawTypes = new Set([
  'available_commands_update',
  'protocol.stderr',
]);

export function buildAgentTimelineModel(events: AgentTimelineEvent[] = []): AgentTimelineModel {
  const visibleEvents: AgentTimelineEvent[] = [];
  const debugEvents: AgentTimelineEvent[] = [];

  for (const event of events) {
    if (event.type === 'assistant_message') continue;
    if (isDebugEvent(event)) {
      debugEvents.push(event);
      continue;
    }
    visibleEvents.push(event);
  }

  visibleEvents.sort(sortTimelineEvents);
  debugEvents.sort(sortTimelineEvents);

  return {
    visibleEvents,
    debugEvents,
    visibleCount: visibleEvents.length,
  };
}

function isDebugEvent(event: AgentTimelineEvent): boolean {
  if (event.type !== 'raw') return false;
  const rawType = typeof event.payload.raw_type === 'string' ? event.payload.raw_type : null;
  if (rawType && debugRawTypes.has(rawType)) return true;
  const sessionUpdate = readSessionUpdate(event.raw);
  return sessionUpdate === 'available_commands_update';
}

function readSessionUpdate(raw: Record<string, unknown> | undefined): string | null {
  const params = raw?.params;
  if (!params || typeof params !== 'object') return null;
  const update = (params as Record<string, unknown>).update;
  if (!update || typeof update !== 'object') return null;
  const sessionUpdate = (update as Record<string, unknown>).sessionUpdate;
  return typeof sessionUpdate === 'string' ? sessionUpdate : null;
}

function sortTimelineEvents(a: AgentTimelineEvent, b: AgentTimelineEvent): number {
  return a.seq - b.seq || a.created_at - b.created_at;
}
```

```ts
export function buildAgentTimelineModel(events: AgentTimelineEvent[] = []): AgentTimelineModel {
  const visibleEvents: AgentTimelineEvent[] = [];
  const debugEvents: AgentTimelineEvent[] = [];

  for (const event of events) {
    if (event.type === 'assistant_message') continue;
    if (isDebugEvent(event)) {
      debugEvents.push(event);
      continue;
    }
    visibleEvents.push(event);
  }

  const mergedVisibleEvents = mergeToolEvents(visibleEvents).sort(sortTimelineEvents);
  debugEvents.sort(sortTimelineEvents);

  return {
    visibleEvents: mergedVisibleEvents,
    debugEvents,
    visibleCount: mergedVisibleEvents.length,
  };
}
```

Add helpers:

```ts
function mergeToolEvents(events: AgentTimelineEvent[]): AgentTimelineEvent[] {
  const result: AgentTimelineEvent[] = [];
  const toolIndexById = new Map<string, number>();

  for (const event of events) {
    const toolId = readToolId(event);
    if (!toolId || (event.type !== 'tool_call' && event.type !== 'tool_result')) {
      result.push(event);
      continue;
    }

    const existingIndex = toolIndexById.get(toolId);
    if (existingIndex === undefined) {
      toolIndexById.set(toolId, result.length);
      result.push(event);
      continue;
    }

    result[existingIndex] = mergeToolEvent(result[existingIndex]!, event);
  }

  return result;
}

function mergeToolEvent(existing: AgentTimelineEvent, incoming: AgentTimelineEvent): AgentTimelineEvent {
  const payload = {
    ...existing.payload,
    ...incoming.payload,
    input: existing.payload.input ?? incoming.payload.input,
    output: incoming.payload.output ?? existing.payload.output,
    kind: incoming.payload.kind ?? existing.payload.kind,
    title: incoming.payload.title ?? existing.payload.title,
    name: incoming.payload.name ?? existing.payload.name,
  };

  return {
    ...existing,
    ...incoming,
    type: incoming.type === 'tool_result' ? 'tool_result' : existing.type,
    status: mergeToolStatus(existing.status, incoming.status),
    title: incoming.title || existing.title,
    payload,
    raw: incoming.raw ?? existing.raw,
    created_at: incoming.created_at,
  };
}

function readToolId(event: AgentTimelineEvent): string | null {
  const id = event.payload.id ?? event.payload.tool_call_id ?? event.payload.toolCallId;
  return typeof id === 'string' && id.trim() ? id : null;
}

function mergeToolStatus(
  existing: AgentTimelineEvent['status'],
  incoming: AgentTimelineEvent['status'],
): AgentTimelineEvent['status'] {
  if (existing === 'failed' || incoming === 'failed') return 'failed';
  if (incoming === 'completed') return 'completed';
  if (existing === 'completed') return 'completed';
  if (incoming === 'delta') return 'delta';
  return incoming;
}
```

```tsx
function FileDiffView({ payload }: { payload: Record<string, unknown> }): JSX.Element {
  const patch = readString(payload.patch) ?? readString(payload.diff) ?? '';
  const path = readString(payload.path);
  const additions = typeof payload.additions === 'number' ? payload.additions : null;
  const deletions = typeof payload.deletions === 'number' ? payload.deletions : null;
  const lines = patch.split('\n');

  return (
    <div className="agent-timeline-diff-wrap">
      <dl className="agent-timeline-kv agent-timeline-diff-meta">
        {path ? (
          <div>
            <dt>文件</dt>
            <dd>{path}</dd>
          </div>
        ) : null}
        {additions !== null ? (
          <div>
            <dt>新增行</dt>
            <dd>{String(additions)}</dd>
          </div>
        ) : null}
        {deletions !== null ? (
          <div>
            <dt>删除行</dt>
            <dd>{String(deletions)}</dd>
          </div>
        ) : null}
      </dl>
      <div className="agent-timeline-diff">
        {lines.map((line, index) => {
          const className = diffLinePattern.test(line) ? (line.startsWith('+') ? 'diff-line is-added' : 'diff-line is-removed') : 'diff-line';
          return (
            <div key={index} className={className}>
              {line || ' '}
            </div>
          );
        })}
      </div>
    </div>
  );
}
```
